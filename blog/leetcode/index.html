# 合并两个有序链表

地址

思路一：

  • 用一个结果链表装载结果
  • 每次移动 l1/l2 中的较小值的那个指针
  • 将结果指针指向较小值的那个节点
  • 后移较小值的指针和结果链表的指针
var mergeTwoLists = function (l1, l2) {
  const prevHead = new ListNode();
  let prevNode = prevHead;
  while (l1 && l2) {
    if (l1.val <= l2.val) {
      prevNode.next = l1;
      l1 = l1.next;
    } else {
      prevNode.next = l2;
      l2 = l2.next;
    }
    prevNode = prevNode.next;
  }
  prevNode.next = l1 || l2;
  return prevHead.next;
};

思路二：

  • 将结果集由 l1/l2 输出
  • 通过移动 l1/l2 的指针不断递归
var mergeTwoLists = function (l1, l2) {
  if (!l1) return l2;
  if (!l2) return l1;
  if (l1.val <= l2.val) {
    l1.next = mergeTwoLists(l1.next, l2);
    return l1;
  } else {
    l2.next = mergeTwoLists(l1, l2.next);
    return l2;
  }
};

if (nums[i] !== nums[i - 1]) {
